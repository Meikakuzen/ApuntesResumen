SELECT COUNT (DISTINCT CLIENTES.NUMERO_DE_SOCIO) AS NUM_CLIENTES
FROM VENTAS
JOIN CLIENTES ON VENTAS.NUMERO_DE_SOCIO = CLIENTES.NUMERO_DE_SOCIO

AND CLIENTES.EDAD >20
